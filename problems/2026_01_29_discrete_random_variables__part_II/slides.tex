\begin{frame}{Correlation Coefficient}
The \textbf{correlation coefficient} $\rho$ shows the strength and direction of the \alert{linear} relationship between two random variables $X$ and $Y$.

\begin{center}
\includegraphics[width=0.8\textwidth]{images/correlation_coefficient.png}	
\end{center}

\vfill 
\pause 
\colorbox{blue!30}{\textbf{Remark.}} We have
\[ |\rho(X,Y)| \leq 1,  \]
with equality if and only if $X$ and $Y$ are linearly related (with probability 1).
\end{frame}

\begin{frame}{Correlation Coefficient}


\colorbox{yellow!30}{\textbf{Poll.}} Can uncorrelated random variables be dependent?
\pause 
\vfill 

	
\colorbox{green!30}{\textbf{Solution.}} Yes. 
\begin{center}
\includegraphics[width=0.8\textwidth]{images/dependent_but_uncorrelated.png}	
\end{center}
The correlation reflects the noisiness and direction of a linear relationship (top row), but not the slope of that relationship (middle), nor many aspects of nonlinear relationships (bottom).

\end{frame}

\begin{frame}{Group exercises - Problem Set 5}
\begin{enumerate}
\item (3.11.16) Let $X$ and $Y$ be independent Bernoulli random variables with parameter $\frac{1}{2}$.  Show that $X+Y$ and $|X-Y|$ are dependent though uncorrelated.
\item (Sec 3.6) Use the Cauchy-Schwarz Inequality to show that the correlation coefficient $\rho$ satisfies $|\rho(X,Y)| \leq 1$ with equality if and only if $Y =mX+b$ with probability 1 for some $m,b \in \R$. (Hint: Apply Cauchy-Schwarz to the variables $X-E[X]$ and $Y-E[Y]$.)
\item (3.7.6) Let $X_1, X_2, \hdots$ be identically distributed random variables with mean $\mu$ and variance $\sigma^2$, and let $N$ be a random variable taking values in the non-negative integers and independent of the $X_i$.  Let $S=X_1+X_2+\cdots+X_N$.  
\begin{itemize}
\item[a.] Show that $\E[S \mid N] = \mu N$, and deduce that $\E[S]=\mu \E[N]$.  
\item[b.] Find $\Var[S]$ using the conditional variance formula $\Var[S]=\E[\Var[S \mid N]] + \Var[\E[S \mid N]]$.	
\end{itemize}

\end{enumerate}   
\end{frame}
